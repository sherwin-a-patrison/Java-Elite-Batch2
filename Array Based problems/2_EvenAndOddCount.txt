Even and Odd Count:

import java.util.*;
public class Main{
	public static void main(String[] args) {
	    Scanner sc=new Scanner(System.in);
	    int n=sc.nextInt(),odd=0,even=0;
	    int arr[]=new int[n];
	    for(int i=0;i<n;i++)
	        arr[i]=sc.nextInt();
	    for(int i=0;i<n;i++){
	        if(arr[i]%2 == 0) even++;
	        else odd++;
	    }
	    System.out.print("Even count:"+even+"\nOdd count:"+odd);
	}
}
