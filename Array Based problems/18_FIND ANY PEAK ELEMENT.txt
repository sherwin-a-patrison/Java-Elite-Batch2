FIND ANY PEAK ELEMENT

class Solution {
    public int peakElement(int[] arr) {
        int a=arr.length;
        if(a==1){
            return 0;
        }
        if(arr[a-1]>arr[a-2]){
            return a-1;}
            if(arr[0]>arr[1]){
                return 0;
            }
        for(int i=1;i<arr.length-1;i++){
            if(arr[i-1]<arr[i]&&arr[i]>arr[i+1]){
                return i;
            }
        }
        return -1;
    }
}
