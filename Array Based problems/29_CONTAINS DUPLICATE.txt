CONTAINS DUPLICATE

class Solution {
    public boolean containsDuplicate(int[] arr) {
        int count=1;
        Arrays.sort(arr);
        for(int i=0;i<arr.length-1;i++){
            if(arr[i]==arr[i+1]){
                count++;
            }
            if(count>=2) {
                return true;
            }
         }
     return false;
    }
}
