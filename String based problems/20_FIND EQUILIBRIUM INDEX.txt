FIND EQUILIBRIUM INDEX

class Solution {
    public static int findEquilibrium(int arr[]) {
        int n=arr.length;
        int total=0,sum=0;
        for(int i=0;i<arr.length;i++){
            total=total+arr[i];
        }
        for(int i=0;i<n;i++){
            total-=arr[i];
            if(total==sum) return i;
            sum+=arr[i];
        }
        return -1;
    }
}
