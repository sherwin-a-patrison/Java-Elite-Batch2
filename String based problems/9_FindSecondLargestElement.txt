Find second largest element:

class Solution {
    public int getSecondLargest(int[] arr) {
        int ans=0,lar=0;
        Arrays.sort(arr);
        for(int i=0;i<arr.length;i++){
            if(lar<arr[i]){
                ans=lar;
                lar=arr[i];
            }
        }
        if(ans==0) return -1;
        return ans;
    }
}
