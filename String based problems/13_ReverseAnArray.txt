Reverse an array:

class Solution {
    public void reverseArray(int arr[]) {
        int n =arr.length, index = 0;
        int ans[] = new int[n];
        for(int i = 0; i < n/2; i++){
            int temp = arr[i];
            arr[i] = arr[n-i-1];
            arr[n-i-1] = temp;
        }
    }
}
