REVERSE SECOND HALF OF ARRAY

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner h=new Scanner(System.in);
		int n=h.nextInt(),mid;
		int arr[]=new int[n];
		for(int i=0;i<n;i++)arr[i]=h.nextInt();
		mid=(n/2);
		for(int i=0;i<(n-mid)/2;i++){
		    int b=arr[mid+i];
		    arr[mid+i]=arr[n-1-i];
		    arr[n-1-i]=b;
		}
	    for(int x:arr)System.out.print(x+" ");
	}
}
